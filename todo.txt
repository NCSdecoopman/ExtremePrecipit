docker run --rm -v "${pwd}:/work" -w /work/perso/v2_modifs texlive/texlive:latest sh -c "latexdiff ../v1_1_preprint/main.tex final/main.tex > diff.tex"

docker run --rm -v "${pwd}:/work" -w /work/perso/v2_modifs texlive/texlive:latest latexmk -pdf -interaction=nonstopmode diff.tex
